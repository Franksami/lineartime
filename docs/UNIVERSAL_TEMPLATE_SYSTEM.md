// Smart package manager detection and configuration
class PackageManagerDetector {
  async detect() {
    const managers = await Promise.all([
      this.checkPnpm(),
      this.checkYarn(), 
      this.checkBun(),
      this.checkNpm()
    ]);
    
    const available = managers.filter(m => m.available);
    const recommended = this.selectRecommended(available);
    
    return {
      available,
      recommended,
      fallback: 'npm' // Always available with Node
    };
  }
  
  async checkPnpm() {
    try {
      const version = await this.getVersion('pnpm --version');
      return {
        name: 'pnpm',
        version,
        available: true,
        score: 95, // Fastest, most efficient
        pros: ['Fastest installs', 'Disk space efficient', 'Strict dependencies'],
        cons: ['Newer tool', 'Some legacy compatibility issues']
      };
    } catch {
      return { name: 'pnpm', available: false };
    }
  }
  
  async checkYarn() {
    try {
      const version = await this.getVersion('yarn --version');
      return {
        name: 'yarn',
        version,
        available: true,
        score: 85, // Good performance, mature
        pros: ['Fast', 'Mature ecosystem', 'Good monorepo support'],
        cons: ['Larger than pnpm', 'Two major versions to choose from']
      };
    } catch {
      return { name: 'yarn', available: false };
    }
  }
  
  async checkBun() {
    try {
      const version = await this.getVersion('bun --version');
      return {
        name: 'bun',
        version,
        available: true,
        score: 90, // Very fast, but newer
        pros: ['Extremely fast', 'All-in-one tool', 'Modern approach'],
        cons: ['Very new', 'Limited Windows support', 'Ecosystem growing']
      };
    } catch {
      return { name: 'bun', available: false };
    }
  }
  
  selectRecommended(available) {
    // Smart recommendation based on project context and platform
    if (available.find(m => m.name === 'pnpm')) {
      return 'pnpm'; // Best overall choice for 2025
    }
    if (available.find(m => m.name === 'bun')) {
      return 'bun'; // Great for new projects
    }
    if (available.find(m => m.name === 'yarn')) {
      return 'yarn'; // Solid choice
    }
    return 'npm'; // Universal fallback
  }
}
