  progressBar(value: number, max: number, width: number = 20): string {
    const filled = Math.round((value / max) * width);
    const empty = width - filled;
    return '█'.repeat(filled) + '░'.repeat(empty);
  }
  
  metricBar(before: number, after: number, type: 'time' | 'size' | 'percent' | 'score'): string {
    const improvement = type === 'time' || type === 'size' 
      ? ((before - after) / before) * 100
      : ((after - before) / before) * 100;
    
    const emoji = improvement > 20 ? '🚀' : improvement > 10 ? '✅' : improvement > 0 ? '📈' : '⚠️';
    
    return `${before} → ${after} (${improvement.toFixed(1)}% ${emoji})`;
  }
}
