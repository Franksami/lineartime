### Resistance Intervention Framework

```typescript
class ResistanceManager {
  async assessResistance(individual: TeamMember): Promise<ResistanceProfile> {
    return {
      level: this.calculateResistanceLevel(individual),
      type: this.identifyResistanceType(individual),
      rootCauses: this.analyzeRootCauses(individual),
      interventions: this.recommendInterventions(individual)
    };
  }
  
  private recommendInterventions(individual: TeamMember): Intervention[] {
    const interventions = [];
    
    if (individual.resistanceType === 'skeptical') {
      interventions.push(
        { type: 'evidence_based', action: 'Show concrete ROI data' },
        { type: 'peer_influence', action: 'Connect with early adopters' },
        { type: 'trial_period', action: 'Low-risk pilot participation' }
      );
    }
    
    if (individual.resistanceType === 'overwhelmed') {
      interventions.push(
        { type: 'support_intensive', action: '1:1 coaching sessions' },
        { type: 'gradual_introduction', action: 'Phased learning approach' },
        { type: 'time_management', action: 'Protected learning time' }
      );
    }
    
    if (individual.resistanceType === 'philosophical') {
      interventions.push(
        { type: 'vision_alignment', action: 'Connect to bigger purpose' },
        { type: 'choice_architecture', action: 'Emphasize voluntary adoption' },
        { type: 'co_creation', action: 'Involve in solution design' }
      );
    }
    
    return interventions;
  }
}
```
