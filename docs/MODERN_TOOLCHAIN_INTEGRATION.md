module_2_production_development:
  duration: "4-6 hours"

  learning_objectives:
    - "Build production-ready applications"

      description: "Add comprehensive testing to React project"
      
      tasks:
        - install_testing_dependencies:
            command: "pnpm add -D vitest @testing-library/react @testing-library/jest-dom @playwright/test"
            explanation: "Vitest for unit tests, Testing Library for React components, Playwright for E2E"
            
        - configure_vitest:
            file: "vitest.config.ts"
            content: |
              import { defineConfig } from 'vitest/config'
              import react from '@vitejs/plugin-react'
              
              export default defineConfig({
                plugins: [react()],
                test: {
                  globals: true,
                  environment: 'jsdom',
                  setupFiles: ['./src/test/setup.ts'],
                  coverage: {
                    provider: 'v8',
                    reporter: ['text', 'html'],
                    thresholds: {
                      global: {
                        statements: 80,
                        branches: 80,
                        functions: 80,
                        lines: 80
                      }
                    }
                  }
                }
              })
            
        - write_first_test:
            file: "src/components/Button.test.tsx"
            content: |
              import { render, screen, fireEvent } from '@testing-library/react'
              import { Button } from './Button'
              
              describe('Button Component', () => {
                it('renders children correctly', () => {
                  render(<Button>Click me</Button>)
                  expect(screen.getByText('Click me')).toBeInTheDocument()
                })
                
                it('handles click events', () => {
                  const handleClick = vi.fn()
                  render(<Button onClick={handleClick}>Click me</Button>)
                  
                  fireEvent.click(screen.getByText('Click me'))
                  expect(handleClick).toHaveBeenCalledTimes(1)
                })
                
                it('applies custom className', () => {
                  render(<Button className="custom">Test</Button>)
                  expect(screen.getByText('Test')).toHaveClass('custom')
                })
              })
            
      validation:
        - "pnpm test (should run and pass)"
        - "pnpm test:coverage (should show coverage report)"
        - "Tests appear in VS Code test explorer"
        
    step_2_deployment_setup:
      duration: "45 minutes" 
      description: "Configure production deployment with monitoring"
      
      platform_options:
        vercel:
          setup: |
            # Install Vercel CLI
            pnpm add -g vercel
            
            # Configure project
            vercel --yes
            
            # Add environment variables
            vercel env add NEXT_PUBLIC_APP_NAME
            
            # Deploy
            vercel --prod
          
          monitoring: "Vercel Analytics + Speed Insights included"
          
        railway:
          setup: |
            # Install Railway CLI
            npm install -g @railway/cli
            
            # Login and deploy
            railway login
            railway init
            railway up
          
          benefits: "No cold starts, persistent storage, competitive pricing"
          
    step_3_quality_gates:
      duration: "30 minutes"
