# Function to show progress with ASCII visualization
show_progress() {
    local step=$1
    local description="$2"
    local progress=$3
    local max_progress=${4:-100}
    
    local percentage=$((progress * 100 / max_progress))
    local filled=$((percentage * 30 / 100))
    local empty=$((30 - filled))
    local bar=$(printf "%-${filled}s" | tr ' ' '█')$(printf "%-${empty}s" | tr ' ' '░')
    
    echo "Step $step: $description"
    echo "Progress: [$bar] $percentage%"
    echo ""
}
